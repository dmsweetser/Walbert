import os

def recursive_file_search(file_path):
    results = []

    for root, dirs, files in os.walk(file_path):
        for file in files:
            file_path_full = os.path.join(root, file)
            try:
                with open(file_path_full, "r") as f:
                    content = f.read()
                    results.append(f"File: {file_path_full}\n```\n{content}\n```\n")
            except Exception as e:
                results.append(f"Error reading file {file_path_full}: {str(e)}")

    return "\n".join(results)
